#define LED = 3;
#define PIR = 2;
#define LDR = A7;
#define LDRTreshhold = 200;
#define dimmedValue = 125;

#define off = 0
#define lit = 1
#define dim = 2

int ledState = off;

void setup() {
  pinMode(LED, OUTPUT); // set strip output
  pinMode(PIR, INPUT); // set pir as input
  pinMode(LED_BUILTIN, OUTPUT); // set builtin led
  digitalWrite(LED_BUILTIN, LOW); // turn off built in led
  attachInterrupt(digitalPinToInterrupt(pin), isr, RISING);
  attachInterrupt(digitalPinToInterrupt(pin), isr, FALLING);
  setStrip(0);
  Serial.begin(115200);
}

void loop() {
    
}

void isr() {
  if(ledState!=off)
    setStrip(255)
}

void isr2() {
  setStrip(0)  
}

void setStrip(char i) {
  analogWrite(LED, i);
  if(i>250)
    ledState = lit
  if(i<250 && i>0)
    ledState = dim
  if(i==0)
    ledState = off
}

bool alreadylit() {
  return analogRead(LDR) < LDRTreshhold ? true : false 
}
